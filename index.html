import { db } from "./firebaseConfig.js"; // Убедись, что импорт базы правильный import { doc, getDoc, setDoc } from "https://www.gstatic.com/firebasejs/9.x.x/firebase-firestore.js"; async function handleAuth(inputName, inputPass) { // 1. Очищаем пробелы, которые могли случайно скопироваться const username = inputName.trim(); const password = inputPass.trim(); // 2. Проверка длины (теперь 123 точно пройдет, так как длина ровно 3) if (username.length < 3 || password.length < 3) { alert("Ник и пароль должны содержать от 3 символов!"); return; } try { // Ссылка на документ в коллекции "users" const userRef = doc(db, "users", username); const userSnap = await getDoc(userRef); if (userSnap.exists()) { // АВТОРИЗАЦИЯ: Пользователь найден const userData = userSnap.data(); if (userData.password === password) { alert("Успешный вход!"); // Здесь твой код перехода в терминал (например, window.location.href = ...) } else { alert("Неверный пароль для этого пользователя!"); } } else { // РЕГИСТРАЦИЯ: Пользователя нет, создаем нового // Прописываем дефолтные поля, как на твоем скриншоте await setDoc(userRef, { password: password, balance: 10000, // Стартовый баланс для нового игрока customRank: 1, exp: 0, farms: 0, isBanned: false, mining: { asic: 0, gpu: 0, heat: 0, powerBill: 0, powerOn: false, quantum: 0 } }); alert("Аккаунт успешно создан! Нажмите Войти еще раз."); } } catch (error) { // Если база отклонит запрос, мы увидим точную причину прямо на экране телефона console.error("Ошибка Firebase:", error); alert("Ошибка базы данных: " + error.message); } }
